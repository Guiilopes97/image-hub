# Configurações do Supabase
# Copie este arquivo para .env e preencha com suas credenciais

# URL do seu projeto Supabase
# Encontre em: Dashboard → Settings → API → Project URL
REACT_APP_SUPABASE_URL=https://seu-projeto.supabase.co

# Chave anônima (anon/public key)
# Encontre em: Dashboard → Settings → API → Project API keys → anon public
REACT_APP_SUPABASE_ANON_KEY=sua-chave-anon-key

# Chave de serviço (service role key) - MANTENHA SECRETO!
# Encontre em: Dashboard → Settings → API → Project API keys → service_role secret
# ⚠️ NUNCA exponha esta chave em código frontend ou em repositórios públicos
SUPABASE_SERVICE_ROLE_KEY=sua-chave-service-role-key
